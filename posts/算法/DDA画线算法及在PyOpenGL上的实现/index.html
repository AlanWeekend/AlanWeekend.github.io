def ddaLine(x0,y0,x1,y1):
    dm=0
    if abs(x1-x0)>abs(y1-y0):
        dm=abs(x1-x0)
    else:
        dm=abs(y1-y0)
    
    dx=(x1-x0)/dm
    dy=(y1-y0)/dm
    
    x=x0+0.5
    y=y0+0.5

    glColor3f(1.0, 1.0, 1.0)
    glPointSize(5)
    for i in range(0,dm):
        glBegin(GL_POINTS)
        glVertex2i(int(x),int(y))
        glEnd()
        x+=dx
        y+=dy

    glClearColor(0.0,0.0,0.0,0.0)

def Reshape(w,h):
    glMatrixMode(GL_PROJECTION)
    glLoadIdentity()
    gluOrtho2D(0.0,GLdouble(w),0.0,GLdouble(h))
